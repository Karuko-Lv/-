```cpp
//初始化线性表中的顺序表
struct SqList{
	int* data;
	int length;//需要一个记录现有长度的变量
	int capacity;
};
void initList(SqList &L,int c){
	L.capacity = c;
	L.data = new int[L.capacity];
	L.length = 0;//刚刚建好的表长度是0
}
int main(){
	SqList L;
	int c = 10;
	initList(L,10);
	return 0;
}
```
