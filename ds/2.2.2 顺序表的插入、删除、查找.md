#define MaxSize 10
typedef struct{
	int data[MaxSize];
	int length;
}SqList;
void InitList(SqList &L){
	L.length = 0;
}
void ListInsert(SqList &L,int i,int e){
//第i个位置插入e这个元素
	L.length+=1;
	//从i位置开始每个元素向后移动一位
	for(int j=L.length;j>=i;j--){
		L.data[j+1]=L.data[j];//别忘了写L.data 不能写成data
	}
	L.data[i]=e;
}
int main(){
	SqList L;
	InitList(L);
	ListInsert(L,1,2);
	return 0;
}

### 删除
![[截屏2026-03-15 11.26.39.png]]
```cpp
#define MaxSize 10
typedef struct{
	int data[MaxSize];
	int length;
}SqList;
void InitList(SqList &L){
	L.length = 0;
}
bool DeleteList(SqList &L,int i){
	if(i>L.length || i<1)
		return false;
	//将i位置之后的移动前一位
	for(int j=i ; j<L.length;j++){
		L.data[j]=L.data[j+1]; 
	}
	L.length--;
	return true;
}
int main(){
	SqList L;
	InitList(L);
	DeleteList(L,3);
	return 0;
}
```

🧚‍♀️由于 e 是实参，而一般情况下实参不可以没有初值; data[5]还在内存里面没有删除；形参是实参的临时拷贝；可是这样最后一个元素还保留着原先的数值啊
修改后的完美代码
```cpp
bool ListDelete(SqList &L, int i, int &e) {
    if (i < 1 || i > L.length) 
        return false;
    
    e = L.data[i - 1];  // 把要死的人先记下来，留遗言
    
    // 把后面的元素依次往前挪一步
    for (int j = i; j < L.length; j++) {
        L.data[j - 1] = L.data[j];
    }
    
    // 🌟 【你要的健壮性代码就在这里】🌟
    // 在小本本 length 减 1 之前，把数组最后那个多出来的“幽灵数据”强行抹除（归零）！
    L.data[L.length - 1] = 0; 
    
    L.length--; // 逻辑长度减 1
    return true;
}
```
🧚‍♀️ c 语言应该是 InitList（&L），SqList（* L），* e=L->data；如果不用循环赋值改用 memcpy 的话复杂度是多少
