### 插入
 ![[截屏2026-03-15 11.01.03.png]]
```cpp
#define MaxSize 10
typedef struct{
	int data[MaxSize];
	int length;
}SqList;
void InitList(SqList &L){
	L.length = 0;
}
void ListInsert(SqList &L,int i,int e){
//第i个位置插入e这个元素
	L.length+=1;
	//从i位置开始每个元素向后移动一位
	for(int j=L.length;j>=i;j--){
		L.data[j+1]=L.data[j];//别忘了写L.data 不能写成data
	}
	L.data[i]=e;
}
int main(){
	SqList L;
	InitList(L);
	ListInsert(L,1,2);
	return 0;
}
```
🤨这里的 i 是位序

### 删除
![[截屏2026-03-15 11.26.39.png]]
```c++
#define MaxSize 10
typedef struct{
	int data[MaxSize];
	int length;
}SqList;
void InitList(SqList &L){
	L.length = 0;
}
bool DeleteList(Sqlist &L,int i){
	if(i>L.length || i<1)
		return false;
	//将i位置之后的移动前一位
	for(int j=i ; j<L.length;j++){
		L.data[j]=L.data[j+1]; 
	}
	L.length--;
	return true;
}
int main(){
	SqList L;
	InitList(L);
	DeleteList(L,3);
}
```
