```cpp
#define MaxSize 10
typedef struct{
	int data[MaxSize];
	int length;
}SqList;
void InitList(SqList &L){
	L.length = 0;
}
bool DeleteList(SqList &L,int i){
	if(i>L.length || i<1)
		return false;
	//将i位置之后的移动前一位
	for(int j=i ; j<L.length;j++){
		L.data[j]=L.data[j+1]; 
	}
	L.length--;
	return true;
}
int main(){
	SqList L;
	InitList(L);
	DeleteList(L,3);
	return 0;
}
```

🧚‍♀️由于 e 是实参，而一般情况下实参不可以没有初值; data[5]还在内存里面没有删除；形参是实参的临时拷贝；可是这样最后一个元素还保留着原先的数值啊
修改后的完美代码
```cpp
bool ListDelete(SqList &L, int i, int &e) {
    if (i < 1 || i > L.length) 
        return false;
    
    e = L.data[i - 1];  // 把要死的人先记下来，留遗言
    
    // 把后面的元素依次往前挪一步
    for (int j = i; j < L.length; j++) {
        L.data[j - 1] = L.data[j];
    }
    
    // 🌟 【你要的健壮性代码就在这里】🌟
    // 在小本本 length 减 1 之前，把数组最后那个多出来的“幽灵数据”强行抹除（归零）！
    L.data[L.length - 1] = 0; 
    
    L.length--; // 逻辑长度减 1
    return true;
}
```

### 查找
#### 按位查找
![[截屏2026-03-15 11.55.03.png]]
存储动态分配的顺序表所需要的内存空间是由 `malloc`  函数申请的一整片的连续空间, 虽然 data 是指针但是同样可以用数组下标的方式访问相应的元素
计算机做了什么❓![[截屏2026-03-15 12.05.06.png]]
指针指向了 `malloc`  函数申请的一整片的连续空间的起始地址，`L.data[0]` 取到的值是从 data 这个指针所指向的地址再往后 6 个字节的内容，用某个类型的**指针加上数组下标**访问数据每次取几个字节和指针所指向的类型有关
🧚‍♀️data[i-1]等价于*(data+i-1)

```cpp
#include <stdio.h>
#define InitSize 10
typedef struct{
	int* data;
	int length;
	int MaxSize;
}SeqList;
void InitList(SeqList &L){
	L.length = 0;
	L.MaxSize = InitSize;
	L.data = new int[L.MaxSize];
}
bool ListInsert(SeqList &L,int i,int e){
	if(i<1 || i>L.length+1){//❌i<0 || i>L.length
		return false;
	}
	if(L.length>=L.MaxSize){//‼️别忘了 且注意MaxSize为结构体中的元素 要加L.
		return false;
	}
	//i-1位往后移动
	for(int j=L.length;j>=i;j--){//别忘了第一个要加int
		L.data[j]=L.data[j-1];
	}
	L.data[i-1]=e;//不是L.data[i]
	L.length++;//不要忘了‼️
	return true;
	
}
int GetElem(SeqList &L,int i){
	return L.data[i-1];/ /⚠️是i-1
}
int main(){
	SeqList L;
	InitList(L);
	int i = 3;
	ListInsert(L,1,1);
	ListInsert(L,2,2);
	ListInsert(L,3,3);
	printf("第%d个元素的值为%d",i,GetElem(L,3));
	return 0;
}
```

#include <iostream>
#include <string>

// 定义一个我们自己的新物种：学生
struct Student {
    int id;               // 学号
    std::string name;     // 名字

    // 🌟 魔法开始：教包工头怎么比较两个 Student 🌟
    // 这句话的意思是：“如果有人对 Student 用 == 符号，你就按我下面的规矩办！”
    bool operator==(const Student& other) const {
        // 我们的规矩：只要学号和名字都一样，就算相等
        return (this->id == other.id) && (this->name == other.name);
    }
};

int main() {
    Student s1 = {1, "Tom"};
    Student s2 = {1, "Tom"};
    Student s3 = {2, "Jerry"};

    // 见证奇迹：现在你可以像比较普通数字一样，直接用 == 比较结构体了！
    if (s1 == s2) {
        std::cout << "s1 和 s2 是同一个人！" << std::endl;
    }

    if (!(s1 == s3)) {
        std::cout << "s1 和 s3 不是同一个人！" << std::endl;
    }

    return 0;
}
