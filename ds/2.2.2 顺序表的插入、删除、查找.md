#define MaxSize 10
typedef struct{
	int data[MaxSize];
	int length;
}SqList;
void InitList(SqList &L){
	L.length = 0;
}
void ListInsert(SqList &L,int i,int e){
//第i个位置插入e这个元素
	L.length+=1;
	//从i位置开始每个元素向后移动一位
	for(int j=L.length;j>=i;j--){
		L.data[j+1]=L.data[j];//别忘了写L.data 不能写成data
	}
	L.data[i]=e;
}
int main(){
	SqList L;
	InitList(L);
	ListInsert(L,1,2);
	return 0;
}
