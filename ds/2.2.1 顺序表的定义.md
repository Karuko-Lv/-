elemtype 需要自己定义数据类型，例如 int、double 等; SqList 即 Sequence 顺序、序列
```cpp
#include<stdio.h>
#define MaxSize 10
typedef struct{
	int data[MaxSize];//静态数组
	int length;//静态顺序表当前的长度
}SqList;
void InitList(SqList &L){
	for(int i = 0; i < MaxSize ; i++){
		L.data[i] = 0;//数据元素初始值设为0
	}
	L.length = 0;//顺序表初始长度为0
}

int main(){
	SqList L;//声明一个顺序表
	InitList(L);//初始化一个顺序表
	//还有一系列操作
	return 0;
}

```
这是一个用于存放整数的顺序表🧚‍♀️这里错了，这里是定义了一个叫 L 的顺序表，main 前面那个只是声明了一个顺序表并把顺序表的类型重定义为 sqlist

```cpp
#include<stdio.h>
#define MaxSize 10
typedef struct{
	int length;//顺序表长度
	int data[MaxSize];//数组申请静态数据表
}SqList;
//初始化若不赋值为0
void InitList(SqList &L){
	L.length = 0;
}
int main(){
	SqList L;
	InitList(L);
	for(int i = 0 ; i < MaxSize ; i++){
		printf("data[%d]=%d\n",i,L.data[i]);
	}
	return 0;
}

```
