elemtype 需要自己定义数据类型，例如 int、double 等; SqList 即 Sequence 顺序、序列
```cpp
#include<stdio.h>
#define MaxSize 10
typedef struct{
	int data[MaxSize];//静态数组
	int length;//静态顺序表当前的长度
}SqList;
void InitList(SqList &L){
	for(int i = 0; i < MaxSize ; i++){
		L.data[i] = 0;//数据元素初始值设为0
	}
	L.length = 0;//顺序表初始长度为0
}

int main(){
	SqList L;//声明一个顺序表
	InitList(L);//初始化一个顺序表
	//还有一系列操作
	return 0;
}

```

动态分配顺序表需要定义一个指向顺序表第一个数据元素的地址的指针

`malloc ` 函数所实现的事情会申请一整片连续的内存空间，这一整片的内存空间一定有起始的内存地址，malloc 函数执行之后会 return 返回一个指向这一片存储空间开始地址的指针

![[截屏2026-03-15 10.10.50.png]] 🧚‍♀️实际运用中，一般会省略强制类型转换，编译器会进行隐式转换
C++可以用 `new` 和 `delete` 这两个关键字来实现类似于 `malloc` 和 `free` 的功能
	`malloc` 函数作用是申请一整片内存空间
```cpp
#include<stdio.h>
#define InitSize 10
typedef struct{
	int* data;//动态顺序表的起始地址
	int length;//动态顺序表的长度
	int MaxSize;//最大长度
}SeqList;
void InitList(SeqList &L){
	L.length=0;
	L.MaxSize = InitSize;
	for(int i=0;i<L.MaxSize;i++){
		L.data[i]=0;
	}
}
int main(){
	SeqList L;
	InitList(L);
	return 0;
}
```
上述代码报错，**💀 野指针狂飙（没批地就建房）** 你是不是只写了 `SeqList L;`，但是**忘了写 `L.data = (int*)malloc(...)`**，就直接迫不及待地去给 `L.data[2]` 赋值了？
```cpp
#include<stdio.h>

void InitList(SeqList &L){
	L.length=0;
	L.MaxSize = InitSize;
	L.data=new int[L.MaxSize];

		L.data[i]=0;

}
int main(){
	SeqList L;
	InitList(L);

}
```

🧚‍♀️直接可以用 realloc 函数进行扩展
```cpp
#define InitSize 10
typedef struct{
	int* data;
	int length;
	int MaxSize;
}SeqList;

void InitList(SeqList &L){
	L.length = 0;
	L.MaxSize = InitSize;
	L.data = new int[L.MaxSize];
}

void IncreaseSize(SeqList &L,int len){
	int *p = L.data;
	L.MaxSize += len;
	L.data = new int[L.MaxSize];
	for(int i = 0;i<L.length;i++){
	//⚠️这里不用L.MaxSize
		L.data = p;
	}
	delete p;
}
int main(){
	SeqList L;
	InitList(L);
	IncreaseSize(L,5);
	return 0;
}

```

🧚‍♀️因为增加元素的方法应该要修改 length，不然还是会出现上节课脏数据那块的情况；L.length = 0 没错，动态顺序表长度此时为零；此操作的顺序为: 建立临时指针指向原空间，data 指针指向新空间，原空间内容通过 p 访问存入 data 相应位置，长度更改，原空间通过 p 指针删除；* p 这个指针用来暂时指向原线性表的所在的空间，然后用 data 这个指针去申请新的空间，申请完了之后，再把每个元素都挪过去。
「PS」free 函数会将 p 指针指向的一整片的空间释放掉，归还给系统
