写一个 c++代码实现设计一个栈，使它可以在 ‭‬‭‬‭‬‭ `O(1)` 时间复杂度内实现 `push` `pop`  `min` 操作（`min` 操作即得到栈中最小的元素）

```c++
#include <iostream>
#include <stack>
#include <stdexcept>

class MinStack {
private:
    std::stack<int> dataStack; // 存放真实数据的栈
    std::stack<int> minStack;  // 存放对应状态下最小值的辅助栈

public:
    // 构造函数
    MinStack() {}
    
    // 入栈操作 O(1)
    void push(int val) {
        dataStack.push(val);
        // 如果辅助栈为空，或者新来的值 <= 当前最小值，则同步压入辅助栈
        // 注意：这里的 <= 非常关键！如果是相等的最小值，也必须压入，防止出栈时错删。
        if (minStack.empty() || val <= minStack.top()) {
            minStack.push(val);
        }
    }
    
    // 出栈操作 O(1)
    void pop() {
        if (dataStack.empty()) {
            throw std::out_of_range("栈为空，无法执行 pop 操作");
        }
        // 如果数据栈弹出的元素正好是当前的最小值，辅助栈也要跟着弹出
        if (dataStack.top() == minStack.top()) {
            minStack.pop();
        }
        dataStack.pop();
    }
    
    // 获取栈顶元素 O(1)
    int top() {
        if (dataStack.empty()) {
            throw std::out_of_range("栈为空，无法获取栈顶元素");
        }
        return dataStack.top();
    }
    
    // 获取栈内最小元素 O(1)
    int getMin() {
        if (minStack.empty()) {
            throw std::out_of_range("栈为空，无最小值");
        }
        return minStack.top(); // 辅助栈的栈顶永远是当前的最小值
    }
};

// --- 测试代码 ---
int main() {
    MinStack minStack;
    
    std::cout << "Push -2, 0, -3" << std::endl;
    minStack.push(-2);
    minStack.push(0);
    minStack.push(-3);
    
    std::cout << "当前最小值: " << minStack.getMin() << " (预期: -3)" << std::endl; // 返回 -3
    
    std::cout << "执行 Pop" << std::endl;
    minStack.pop(); // 弹出 -3
    
    std::cout << "当前栈顶: " << minStack.top() << " (预期: 0)" << std::endl;      // 返回 0
    std::cout << "当前最小值: " << minStack.getMin() << " (预期: -2)" << std::endl; // 返回 -2
    
    return 0;
}
```
